#include <iostream>
using namespace std;
int main()
{
int i, j, aux;
int v2[] = {3,2,1,5,4}; //vector desordenado NARRAY = 5

	for (i=1; i<5; i++)
	{
		aux = v2[i];
		j = i;
		while ((j>0) && (v2[j-1] > aux))
		{
			v2[j] = v2[j-1];
			j = j - 1;
		}
		v2[j] = aux;
	}

cout<<"Ordenamiento Secuencial de Insertion Sort  ";
 for(i=0;i<5;i++){
        cout<<" ";
           cout<<v2[i];
    }
return 0;
}
